% Section = full-width solid banner: the strongest, unmistakable "new topic" cue.
% (Implemented as a standalone command rather than \titleformat, so the
%  colored banner can safely span the whole title without brace-splitting
%  tricks against titlesec's internals.)
\newcommand{\sectionbanner}[1]{%
  \par\vspace{1.4ex}%
  \refstepcounter{section}%
  \label{sec:\arabic{section}}%
  \noindent\colorbox{secRule}{\parbox[c][3.2ex][c]{\dimexpr\linewidth-2\fboxsep\relax}{\color{white}\bfseries\sffamily\fontsize{10.5pt}{12.5pt}\selectfont\thesection\quad #1}}%
  \par\vspace{0.9ex}%
}
\renewcommand{\section}{\sectionbanner}

% Running footer: current section (left) + page number (right) for fast wayfinding.
% Built from \pageref, not TeX marks: multicol's column-splitting delivers marks
% out of page order, so \leftmark/\rightmark cannot be trusted here.
\newcommand{\tsCheckCarry}[2]{\ifnum\getpagerefnumber{sec:#1}<\value{page}\relax\gdef\tsCarry{#1\quad #2}\fi}
\newcommand{\tsCheckLast}[2]{\ifnum\getpagerefnumber{sec:#1}>\value{page}\relax\else\gdef\tsLast{#1\quad #2}\fi}
