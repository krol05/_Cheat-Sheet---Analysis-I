\subsection*{Rules of derivation (holds only if $f^{-1}$ exists and $f$ continuous.)}
\begin{itemize}[noitemsep, topsep=0pt, leftmargin=*]
\item \textbf{Sum:}
\(
(f+g)'=f'+g'
\)
\item \textbf{Product:}
\(
(fg)'=f'g+fg'
\)
\item \textbf{Chain rule:}
\(
(g\circ f)'=g'(f)f'
\)
\item \textbf{Quotient rule:}
\(
\left(\frac fg\right)'=\frac{f'g-fg'}{g^2}
\)
\item \textbf{Derivative of the Inverse:}
\(
(f^{-1})'(y_0)=\frac1{f'(x_0)},\quad y_0=f(x_0)
\)
\end{itemize}


\subsection*{Rules of integration}
\begin{itemize}[noitemsep, topsep=0pt, leftmargin=*]
\item \textbf{Linearity of Integration:}
\(
\int(f+g)=\int f+\int g \text{ and }
\int cf=c\int f
\)
\item \textbf{Integration by Parts:}
\(
\int f'g=fg-\int fg'
\)
\item \textbf{Substitution:}
\(
\int g(f(x))f'(x)\,dx=\int g(y)\,dy
\)
\end{itemize}

\subsection*{Logarithms}
\begin{itemize}[noitemsep, topsep=0pt, leftmargin=*]
\item \textbf{Change of base:} $\log_a x=\frac{\ln x}{\ln a}$
\item \textbf{Power rule:} $\log_a(x^y)=y\log_a x$
\item \textbf{Product rule:} $\log_a(xy)=\log_a x+\log_a y$
\item \textbf{Quotient rule:} $\log_a\left(\frac xy\right)=\log_a x-\log_a y$
\item \textbf{Logarithm 1:} $\log_a\left(1\right)=0 \ \forall a \in \mathbb{N}$ (negative undefined)
\end{itemize}

\subsection*{Trigonometry}
\begin{itemize}[noitemsep, topsep=0pt, leftmargin=*]
\item \textbf{Tangent and cotangent:} $\tan x=\frac{\sin x}{\cos x},\quad \cot x=\frac{\cos x}{\sin x}$
\item \textbf{Pythagorean identity:} $\sin^2x+\cos^2x=1$
\item \textbf{Double-angle sine:} $\sin(2x)=2\sin x\cos x$
\item \textbf{Double-angle cosine:} $\cos(2x)=\cos^2x-\sin^2x$
\item \textbf{Angle addition sine:} $\sin(x+y)=\sin x\cos y+\cos x\sin y$
\item \textbf{Angle addition cosine:} $\cos(x+y)=\cos x\cos y-\sin x\sin y$
\item \textbf{Even/Odd symmetry:} $\cos(x)=\cos(-x)$ and $\sin(-x)=-\sin(x)$
\item \textbf{Reflection at $\pi/2$:} $\cos(\pi-x)=-\cos(x)$ and $\sin(\pi-x)=\sin(x)$
\end{itemize}

\subsection*{Hyperbolic}
\begin{itemize}[noitemsep, topsep=0pt, leftmargin=*]
\item \textbf{Hyperbolic sine:} $\sinh(x):=\frac{e^x-e^{-x}}{2}:\mathbb{R}\to\mathbb{R}$
\item \textbf{Hyperbolic cosine:} $\cosh(x):=\frac{e^x+e^{-x}}{2}:\mathbb{R}\to[1,\infty)$
\item \textbf{Hyperbolic tangent:} $\tanh(x):=\frac{\sinh(x)}{\cosh(x)}=\frac{e^x-e^{-x}}{e^x+e^{-x}}:\mathbb{R}\to(-1,1)$
\item \textbf{Inverse hyperbolic sine:} $\operatorname{arsinh}x=\ln(x+\sqrt{x^2+1})$
\item \textbf{Inverse hyperbolic cosine:} $\operatorname{arcosh}x=\ln(x+\sqrt{x^2-1})$
\item \textbf{Inverse hyperbolic tangent:} $\operatorname{artanh}x=\frac12\ln\frac{1+x}{1-x}$
\end{itemize}

\subsection*{Complement trick}
\[
\sqrt{ax+b}-\sqrt{cx+d}
=
\frac{ax+b-(cx+d)}
{\sqrt{ax+b}+\sqrt{cx+d}}
\]

\subsection*{Integration by parts}
Should we get unavoidable cycle, where we have to integrate the same thing again, we may simply add the integral to both sides, and we thus have $2$ times the integral on the left side and then finish the integration by parts on the right hand side and in the end divide by the factor up front to get the result.
